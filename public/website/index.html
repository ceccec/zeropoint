Begin your journey from zero to infinite

Add to Gemfile 
# Add to your Gemfile
gem 'zeropoint'

# Install
bundle install

# Generate configuration
rails generate zeropoint:install
Basic Usage 
# Simple streaming controller
class UsersController < ApplicationController
  include Zeropoint::ControllerConcern
  
  def index
    @users = zeropoint_stream(
      filters: { active: true },
      query: params[:q],
      streaming: true
    )
  end
  
  def realtime
    zeropoint_realtime(filters: { active: true }) do |user|
      response.stream.write("data: #{user.to_json}\n\n")
    end
  end
end
