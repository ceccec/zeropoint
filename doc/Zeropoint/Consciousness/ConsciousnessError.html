def category
  case @original_error.class.name
  when /GraphQL/
    'graphql'
  when /Database/
    'database'
  when /Network/
    'network'
  when /Validation/
    'validation'
  else
    'general'
  end
end

def severity
  case @original_error.class.name
  when /Timeout/, /Connection/
    8
  when /Validation/, /Authorization/
    6
  when /NotFound/
    4
  else
    5
  end
end
