def validate_patterns
  POSITION_1_CIRCUIT.length == 6 &&
  POSITION_2_CIRCUIT.length == 6 &&
  GAP_POSITIONS.all? { |pos| pos % 3 == 0 } &&
  MAGNETIC_EFFICIENCY_GAIN > 0 &&
  COPPER_REDUCTION_MIN > 0
end
