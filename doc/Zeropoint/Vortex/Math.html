def axis_state(n)
  case digital_root(n)

  when 6 then :axis_6
  when 9 then :axis_9
  else :off_axis
  end
end

def digital_root(n)
  n = n.to_i.abs
  return 9 if n == 0
  r = n % 9
  r.zero? ? 9 : r
end

def fibonacci(n)
  return 0 if n <= 0
  return 1 if n == 1
  a = 0
  b = 1
  (2..n).each { a, b = b, a + b }
  b
end

def golden_angle(n = 1)
  (360.0 * (1 - 1.0 / phi) * n) % 360

def prime?(n)
  n = n.to_i
  return false if n < 2
  (2..::Zeropoint::Math.sqrt(n)).none? { |i| n % i == 0 }
end

def primes(k)
  primes = []
  n = 2
  while primes.length < k
    primes << n if prime?(n)
    n += 1
  end
  primes
end

def toroidal_halving(n)
  # Inverse of doubling in mod 9 (finds a number x such that digital_root(x*2) == n)
  (1..9).find { |x| toroidal_doubling(x) == digital_root(n) } || 0
end
