def create(type, **options)
  case type.to_sym
  when :rodin
    Rodin.new(**options)
  when :toroidal
    Toroidal.new(**options)
  when :regular
    Regular.new(**options)
  else
    raise ArgumentError, "Unknown coil type: #{type}. Available types: #{available_types.join(', ')}"
  end
end
